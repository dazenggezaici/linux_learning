补充预定义变量：$!  # 最后一个放在后台进程的进程号
expr 1 + 1  # 这里加号两边都需要加空格
	2
expr 1 \* 2  # 这里乘号需要加转义字符

echo $[1+1]  # 中括号里也可以写变量如：$[a+b]
	2

let命令：这条命令与expr类似，但不会把结果显示出来，对象主要是变量
	通常用于修改变量或者定义变量
	a=0
	let a++
	let a+=1
	let a=a+1
	let a%=3 ; echo $a
		0
以上工具只支持整数运算
bc交互式运算
在交互式界面，scale=x，可以定义显示小数点后x位
该工具在脚本中一般用非交互式的运算
echo 'scale=3;10/3' | bc  # 非交互式

条件测试
'test 表达式'  或者这样写 [ 表达式 ]  # 表达式两边都需要一个空格，如
	[ $USER == 'root' ]
	
&&：只有当&&前面的命令执行成功后，&&后的语句才会继续执行
||：只有当&&前面的命令执行失败后，&&后的语句才会继续执行

[ -z $abc ]  # 判断abc变量是否为空，空值就返回0

-eq等于
-ne不等于
-gt大于
-ge大于等于
-lt小于
-le小于等于

-e判断对象是否存在
-f判断对象是否是文件
-d判断对象是否是目录
-r,-w,-x判断当前用户是否有读写执行权限
	注：管理员root默认对所有文件都有读写权限，但不会默认有执行权限
RANDOM随即值
$[RANDOM%10]  # 这里就可以生成0-9的随机数字，跟py的
